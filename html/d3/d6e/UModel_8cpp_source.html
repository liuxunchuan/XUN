294 inline double UModel::K(int i, double Temp, double AV){
295  int type, TR=0;
296  double ALF, BET, GAM, ZETA;
297  double DTMIN,DT;
298 
299  TR = 0;
300  DTMIN = 1.E100;
301  if(this->REAC[i].NTR>1)
302  for(int j=0; j<this->REAC[i].NTR ;j++){
303  if(this->REAC[i].TINT[j] >= Temp && this->REAC[i].TINT[j]<Temp){
304  TR = j;
305  break;
306  }
307  DT = min( abs(this->REAC[i].TINT[j]-Temp), abs(this->REAC[i].TEND[j]-Temp));
308  if (DT < DTMIN){
309  TR = j;
310  DTMIN = DT;
311  }
312  }
313  ALF = this->REAC[i].ALF[TR];
314  BET = this->REAC[i].BET[TR];
315  GAM = this->REAC[i].GAM[TR];
316 
317  //if(this->REAC[i].TINT[0]>30 and this->REAC[i].GAM < 0) return 0.;
318 
319  if(this->REAC[i].type == 1) //COSMIC RAY PARTICLE RATE
320  return ALF * this->ZETA;
321  else if(this->REAC[i].type == 2) //COSMIC RAY PHOTO-RATE
322  return ALF * pow(Temp/300.0,BET)* GAM * this->ZETA/(1.0-this->ALBEDO);
323  else if(this->REAC[i].type == 3) //PHOTO-REACTION RATE
324  return ALF*exp(-GAM*AV);
325  else if(this->REAC[i].type == 10){
326  return 0.0;
327  }else
328  return ALF*pow(Temp/300.0,BET)*exp(-GAM/Temp);
